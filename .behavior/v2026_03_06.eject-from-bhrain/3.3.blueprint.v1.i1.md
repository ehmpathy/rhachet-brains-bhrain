## architecture

```
┌─────────────────────────────────────────────────────────┐
│ consumer code                                           │
│                                                         │
│   // discover repl+atom via composite slug              │
│   const brain = await genContextBrain({                 │
│     choice: 'bhrain/arch1@claude/sonnet'                │
│   });                                                   │
│                                                         │
│   // use directly — atom is baked in                    │
│   brain.ask({ prompt, schema, role });                  │
│   brain.act({ prompt, schema, role });                  │
│                                                         │
└────────────────────────┬────────────────────────────────┘
                         │ BrainRepl interface (atom baked in)
                         ▼
┌─────────────────────────────────────────────────────────┐
│ genBrainRepl({ slug: 'bhrain/arch1@claude/sonnet' })    │
│   ├─ parses composite slug: $repl@$atom                │
│   │     repl = 'bhrain/arch1'                          │
│   │     atom = 'claude/sonnet'                         │
│   ├─ discovers atom via rhachet brain discovery        │
│   ├─ creates BrainArch1Repl with discovered atom       │
│   ├─ ask/act invoke invokeBrainArch1                   │
│   └─ transforms BrainArch1LoopResult to BrainOutput    │
└────────────────────────┬────────────────────────────────┘
                         │ internal implementation
                         ▼
┌─────────────────────────────────────────────────────────┐
│ invokeBrainArch1({ repl, userInput }, context)   [←]    │
│   ├─ repl.atom = atom from slug (baked in at creation) │
│   ├─ validates input                                    │
│   ├─ merges toolboxes                                   │
│   ├─ runs agentic loop via runBrainArch1Loop           │
│   └─ returns BrainArch1LoopResult                      │
└─────────────────────────────────────────────────────────┘
```

**composite slug pattern**: `$repl@$atom`
- `bhrain/arch1@claude/sonnet` = BrainArch1 repl with claude sonnet atom
- `bhrain/arch1@xai/grok` = BrainArch1 repl with xai grok atom
- `bhrain/arch1` alone = invalid (repl requires atom to function)

**key insight**: repl slug without atom is unusable — composite slug bakes them together

## transformations (post-ejection)

the ejected BrainArch1 code requires transformation to integrate with rhachet ecosystem:

| from (roles repo) | to (brains repo) | why |
|-------------------|------------------|-----|
| `BrainArch1Actor` | `BrainArch1Repl` | clarifies this is the repl implementation |
| `BrainArch1Atom` | `BrainAtom` (from rhachet) | use standard rhachet atom interface |
| direct atom creation | parsed from composite slug | atom baked in at repl creation |

**key insight**: BrainArch1Repl.atom accepts `BrainAtom` directly — no adapter needed

**composite slug pattern**: `bhrain/arch1@$atom`
- repl slug alone (`bhrain/arch1`) is invalid — requires atom
- atom is discovered and baked in at genBrainRepl creation time
- callers use `genContextBrain({ choice: 'bhrain/arch1@claude/sonnet' })`

**peer dependency pattern**: no direct dependency on rhachet-brains-* packages
- genBrainRepl discovers atom via rhachet's `genBrainAtom({ slug: atomSlug })`
- this package only declares peerDependencies on brain suppliers
- tests use devDependencies for integration tests

## genBrainRepl contract

```ts
/**
 * .what = factory for BrainRepl that wraps BrainArch1
 * .why = exposes rhachet's BrainRepl interface; BrainArch1 powers it internally
 *
 * slug pattern: bhrain/arch1@$atom
 *   - bhrain/arch1@claude/sonnet
 *   - bhrain/arch1@xai/grok
 *   - bhrain/arch1 alone is invalid (repl requires atom)
 *
 * usage via brain discovery:
 *   const brain = await genContextBrain({ choice: 'bhrain/arch1@claude/sonnet' });
 *   await brain.ask({ prompt, schema, role });
 */
export const genBrainRepl = (input: {
  slug: `bhrain/arch1@${string}`;
}): BrainRepl => {
  // parse composite slug
  const [replSlug, atomSlug] = parseCompositeSlug(input.slug);
  if (!atomSlug) throw new BadRequestError('repl slug requires @$atom suffix', { slug: input.slug });

  // discover atom via rhachet brain discovery
  const atom = await genBrainAtom({ slug: atomSlug });

  return new BrainRepl({
    repo: 'bhrain',
    slug: input.slug,
    spec: { /* from config */ },
    description: `bhrain agentic brain via BrainArch1, atom: ${atomSlug}`,
    ask: async (askInput) => invokeBrainArch1WithMode({ mode: 'ask', replInput: askInput, atom }),
    act: async (actInput) => invokeBrainArch1WithMode({ mode: 'act', replInput: actInput, atom }),
  });
};

// internal: translate BrainRepl interface to BrainArch1 calls
const invokeBrainArch1WithMode = async (
  input: { mode: 'ask' | 'act'; replInput: BrainReplInput; atom: BrainAtom },
) => {
  // atom is baked in at creation time (from composite slug)
  const repl: BrainArch1Repl = {
    atom: input.atom,
    role: { systemPrompt: buildSystemPrompt(input.replInput.role.briefs) },
    toolboxes: input.mode === 'ask' ? READONLY_TOOLBOXES : READWRITE_TOOLBOXES,
    constraints: { maxIterations: 50, maxTokens: 200000 },
    memory: null,
    permission: null,
  };

  // invoke BrainArch1
  const result = await invokeBrainArch1(
    { repl, userInput: input.replInput.prompt },
  );

  // transform to BrainOutput
  return {
    output: parseOutput(result.finalResponse, input.replInput.schema),
    metrics: extractMetrics(result),
    episode: buildEpisode(result),
  };
};

// helper: parse $repl@$atom slug
const parseCompositeSlug = (slug: string): [string, string | null] => {
  const atIndex = slug.indexOf('@');
  if (atIndex === -1) return [slug, null];
  return [slug.slice(0, atIndex), slug.slice(atIndex + 1)];
};
```
